function updateImage(){ let style = val("style"); let outfit = val("outfit"); let img = document.getElementById("previewImg"); // DEFAULT (REALISTIC) let url = "https://images.unsplash.com/photo-1529626455594-4ff0802cfb7e"; // STYLE SWITCH (using more reliable sources) if(style.includes("Chibi")){ url = "https://placehold.co/600x800/png?text=Chibi+Style"; } else if(style.includes("Anime")){ url = "https://placehold.co/600x800/png?text=Anime+Style"; } else if(style.includes("Bratz")){ url = "https://placehold.co/600x800/png?text=Bratz+Style"; } else{ // REALISTIC + OUTFIT SWITCH if(outfit.includes("hoodie") || outfit.includes("Streetwear")){ url = "https://images.unsplash.com/photo-1503342217505-b0a15ec3261c"; } else if(outfit.includes("gown")){ url = "https://images.unsplash.com/photo-1520975922284-9e0f3a5e54c1"; } } // FORCE REFRESH img.src = url + "&t=" + new Date().getTime(); // 🔥 FALLBACK (if image fails) img.onerror = function(){ img.src = "https://placehold.co/600x800/png?text=Preview"; }; }
